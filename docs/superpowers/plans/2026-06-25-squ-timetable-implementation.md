# squ-timetable Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a PocketBase-backed React SPA for university timetable viewing, replacing the existing Google Sheets-backed app.

**Architecture:** Single-page React app using TanStack Router + TanStack Query + PocketBase JS SDK. Static files served via PocketBase's `pb_public/`. All data fetched client-side from PocketBase API with `expand` for relations.

**Tech Stack:** React 19, TypeScript 5.9, Vite 8, MUI 9, TanStack Router, TanStack Query, Zustand, PocketBase 0.27, ECharts, framer-motion

## Global Constraints

- Target directory: `C:\Users\kalho\Desktop\github\squ-timetable`
- PocketBase server: `https://tt.earthsciences.science/`
- Superuser token from `.env` as `PB_SUPERUSER_TOKEN`
- Frontend URL from `.env` as `VITE_PB_URL`
- Build output goes to `pocketbase/pb_public/`
- All collections accessible for public read (`listRule/viewRule: ""`)
- Write rules: `@request.auth.id != "" && @request.auth.person.isAdmin = true`
- PocketBase JS SDK v0.27 for browser use
- No server-side proxy; pure SPA

async function createCollection(def: CollectionDef, retries = 0): Promise<string | null> {
  try {
    const res = await api('/collections', {
      method: 'POST',
      body: JSON.stringify(def),
    })
    if (!res.ok) {
      const body = await res.text()
      // If collection already exists, return null
      if (res.status === 400 && body.includes('already exists')) return null
      throw new Error(`Failed to create ${def.name}: ${res.status} ${body}`)
    }
    const data = await res.json()
    console.log(`  Created: ${def.name} (${data.id})`)
    return data.id
  } catch (err) {
    if (retries < 3) {
      await new Promise(r => setTimeout(r, 1000))
      return createCollection(def, retries + 1)
    }
    throw err
  }
}

async function main() {
  console.log('Creating collections...\n')

  // We need to do this in phases because earlier collections need to exist
  // before we reference them in relation fields.
  // Actually, PocketBase allows deferred relation resolution by name,
  // but we need the actual collection IDs for the seed script.
  // For simplicity, we use static placeholder IDs in the field definitions
  // which PB will resolve. But actually PB requires real collection IDs.
  //
  // SIMPLER APPROACH: Create all collections without relations first,
  // then update them with the correct relation fields.

  // Step 1: Delete existing collections if they exist (order matters for cascade)
  console.log('Step 1: Cleaning up existing collections...')
  const collectionsToDelete = [
    'sessions', 'sections', 'exams', 'courses', 'instructors',
    'departments', 'halls', 'buildings', 'semesters', 'colleges',
  ]
  const existing = await api('/collections?perPage=100')
  const existingData = await existing.json()
  if (existingData?.items) {
    for (const col of existingData.items) {
      if (collectionsToDelete.includes(col.name)) {
        await api(`/collections/${col.id}`, { method: 'DELETE' })
        console.log(`  Deleted: ${col.name}`)
      }
    }
  }

  // Step 2: Create collections without relation fields
  // Each collection gets a minimal create, then we update with fields
  // Actually, PocketBase lets us create collections with the full field definitions
  // in one shot. But the relation fields need to reference other collection IDs
  // which don't exist yet.
  //
  // BEST APPROACH: Create all collections with their basic fields first,
  // then add relation fields as updates.

  console.log('\nStep 2: Creating basic collections...')
  const basicDefs = COLLECTIONS.map(c => ({
    ...c,
    fields: c.fields.filter(f => f.type !== 'relation'),
  }))

  // Actually, this is getting complex. Let me use a simpler approach:
  // Create collections one by one in dependency order, storing the returned IDs,
  // then create the full field definitions in one shot per collection.

  console.log('This script requires manual adjustment for your PB version.')
  console.log('Collections will be created in dependency order.')

  // Always create no-dependency collections first
  for (const name of ['colleges', 'buildings', 'semesters']) {
    const def = COLLECTIONS.find(c => c.name === name)!
    await createCollection(def)
  }

  // Wait a moment for PB to index
  await new Promise(r => setTimeout(r, 500))

  // Fetch the created collection IDs
  const listRes = await api('/collections?perPage=100')
  const listData = await listRes.json()
  const created: Record<string, string> = {}
  if (listData?.items) {
    for (const col of listData.items) {
      if (collectionsToDelete.includes(col.name)) {
        created[col.name] = col.id
      }
    }
  }

  console.log('\nCreated collection IDs:', JSON.stringify(created, null, 2))

  // Now create remaining collections with resolved relation IDs
  function resolveDef(name: string): CollectionDef | null {
    const def = COLLECTIONS.find(c => c.name === name)
    if (!def) return null

    const collectionIdPattern = /pbc_\?\?_(\w+)/g

    const resolvedFields = def.fields.map(f => {
      if (f.type === 'relation' && typeof f.collectionId === 'string') {
        const match = f.collectionId.match(/pbc_\?\?_(\w+)/)
        if (match && created[match[1]]) {
          return { ...f, collectionId: created[match[1]] }
        }
      }
      return f
    })

    return { ...def, fields: resolvedFields }
  }

  // departments (depends on colleges)
  const deptDef = resolveDef('departments')
  if (deptDef) await createCollection(deptDef)

  // Re-fetch IDs
  await new Promise(r => setTimeout(r, 500))
  const listRes2 = await api('/collections?perPage=100')
  const listData2 = await listRes2.json()
  if (listData2?.items) {
    for (const col of listData2.items) {
      if (collectionsToDelete.includes(col.name)) {
        created[col.name] = col.id
      }
    }
  }

  console.log('\nUpdated collection IDs:', JSON.stringify(created, null, 2))

  // Now remaining collections
  for (const name of ['instructors', 'courses', 'halls', 'sections', 'sessions', 'exams']) {
    const def = resolveDef(name)
    if (def) await createCollection(def)
  }

  console.log('\nDone! All collections created.')
}

main().catch(err => {
  console.error('Fatal:', err)
  process.exit(1)
})

function flattenSession(session: SessionRecord): ScheduleRow {
  const course = session.expand?.section?.expand?.course
  const section = session.expand?.section
  const semester = section?.expand?.semester
  const instructors = session.expand?.instructor ?? []
  const hall = session.expand?.hall
  const building = hall?.expand?.building
  const department = course?.expand?.department

  return {
    id: session.id,
    courseCode: course?.code ?? '',
    courseName: course?.name ?? '',
    sectionNumber: section?.number ?? 0,
    creditHours: course?.credit_hours ?? 0,
    level: course?.level ?? 'UG',
    day: session.day,
    startTime: session.start_time,
    endTime: session.end_time,
    hallName: hall?.name ?? '',
    buildingName: building?.name ?? '',
    instructorNames: instructors.map(i => i.name),
    instructorIds: instructors.map(i => i.id),
    teachingMethod: session.teaching_method ?? '',
    departmentName: department?.name ?? '',
    collegeName: department?.expand?.college?.name ?? '',
    semesterId: semester?.id ?? '',
    semesterLabel: semester ? `${semester.term} ${semester.year}` : '',
  }
}

export function useSessions(semesterId?: string) {
  const filter = semesterId
    ? `section.semester = '${semesterId}'`

        sort: 'day,start_time',
        filter: filter || undefined,
        expand: 'section.course.department.college,section.semester,instructor,hall.building',
        perPage: 10000,
      })
      return sessions.map(flattenSession)
    },
    staleTime: 30 * 60 * 1000,
  })

// src/stores/studentTableStore.ts
import { create } from 'zustand'
import { persist } from 'zustand/middleware'
import type { ScheduleRow } from '../lib/queries/types'

interface StudentState {
  chosenSections: ScheduleRow[]
  allowConflicts: boolean
  addSection: (row: ScheduleRow) => void
  removeSection: (id: string) => void
  setAllowConflicts: (v: boolean) => void
  clear: () => void
}

function hasConflict(rows: ScheduleRow[]): boolean {
  for (let i = 0; i < rows.length; i++) {
    for (let j = i + 1; j < rows.length; j++) {
      const a = rows[i], b = rows[j]
      if (a.day !== b.day) continue
      if (a.startTime < b.endTime && b.startTime < a.endTime) return true
    }
  }
  return false
}

export const useStudentTableStore = create<StudentState>()(
  persist(
    (set, get) => ({
      chosenSections: [],
      allowConflicts: false,
      addSection: (row) => set((s) => {
        if (s.chosenSections.some(r => r.id === row.id)) return s
        return { chosenSections: [...s.chosenSections, row] }
      }),
      removeSection: (id) => set((s) => ({ chosenSections: s.chosenSections.filter(r => r.id !== id) })),
      setAllowConflicts: (allowConflicts) => set({ allowConflicts }),
      clear: () => set({ chosenSections: [] }),
    }),
    { name: 'student-timetable' },
  ),
)

// src/pages/faculty/InstructorTimetable.tsx
import { Box, Typography, Autocomplete, TextField, ToggleButtonGroup, ToggleButton, useMediaQuery } from '@mui/material'
import { useState, useMemo } from 'react'
import { useSessions, useActiveSemester } from '../../lib/queries'
import { WeeklySchedule } from '../../components/WeeklySchedule'
import { FinalExamSchedule } from '../../components/FinalExamSchedule'
import { MyCustomSpinner } from '../../components/MyCustomSpinner'
import type { ScheduleRow } from '../../lib/queries/types'

// Extract unique instructors from sessions
function getInstructors(rows: ScheduleRow[]): { id: string; name: string }[] {
  const map = new Map<string, string>()
  rows.forEach(r => r.instructorIds.forEach((id, i) => map.set(id, r.instructorNames[i])))
  return Array.from(map.entries()).map(([id, name]) => ({ id, name })).sort((a, b) => a.name.localeCompare(b.name))
}

export default function InstructorTimetable() {
  const { data: activeSemester } = useActiveSemester()
  const { data: sessions, isLoading } = useSessions(activeSemester?.id)
  const [selectedInstructors, setSelectedInstructors] = useState<string[]>([])
  const [view, setView] = useState<'weekly' | 'exam'>('weekly')
  const isMobile = useMediaQuery('(max-width:600px)')

  const instructors = useMemo(() => sessions ? getInstructors(sessions) : [], [sessions])

  const filtered = useMemo(() => {
    if (!sessions || !selectedInstructors.length) return []
    return sessions.filter(r => r.instructorIds.some(id => selectedInstructors.includes(id)))
  }, [sessions, selectedInstructors])

  if (isLoading) return <MyCustomSpinner message="Loading..." />

  return (
    <Box>
      <Typography variant="h5" fontWeight={700} mb={2}>Instructor Schedule</Typography>

      <Autocomplete
        multiple
        size="small"
        options={instructors}
        getOptionLabel={(o) => o.name}
        value={instructors.filter(i => selectedInstructors.includes(i.id))}
        onChange={(_, v) => setSelectedInstructors(v.map(i => i.id))}
        renderInput={(p) => <TextField {...p} label="Select instructors..." />}
        sx={{ mb: 2, maxWidth: 500 }}
      />

      <ToggleButtonGroup value={view} exclusive onChange={(_, v) => v && setView(v)} size="small" sx={{ mb: 2 }}>
        <ToggleButton value="weekly">Weekly</ToggleButton>
        <ToggleButton value="exam">Exams</ToggleButton>
      </ToggleButtonGroup>

      {selectedInstructors.length > 0 ? (
        view === 'weekly' ? <WeeklySchedule rows={filtered} /> : <FinalExamSchedule exams={[]} />
      ) : (
        <Typography color="text.secondary">Select an instructor to view their schedule.</Typography>
      )}
    </Box>
  )
}
